def parse_note_detail_from_page(page, note_url: str) -> dict:
    page.goto(note_url, wait_until="domcontentloaded")
    page.wait_for_timeout(3000)

    title = ""
    desc = ""
    content = ""
    tags = []
    cover = ""

    # title
    try:
        title = page.locator("meta[property='og:title']").get_attribute("content") or ""
    except:
        pass

    # desc
    try:
        desc = page.locator("meta[name='description']").get_attribute("content") or ""
    except:
        pass

    # cover（关键修复）
    try:
        cover = page.locator("meta[property='og:image']").get_attribute("content") or ""
    except:
        pass

    # 正文（优化）
    try:
        paragraphs = page.locator("p").all_inner_texts()
        content = "\n".join(paragraphs[:20])
    except:
        pass

    # 标签（优化）
    try:
        tag_elements = page.locator("a").all_inner_texts()
        tags = [t.strip() for t in tag_elements if t.strip().startswith("#")][:10]
    except:
        pass

    return {
        "title": title,
        "desc": desc,
        "content": content,
        "tags": tags,
        "cover_url": cover,
        "note_url": note_url,
    }
